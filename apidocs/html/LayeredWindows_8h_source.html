00066     if ((hUser32 != NULL) && g_enableLayeredWindows &&
00067         (*(FARPROC*)&g_pfnSetLayeredWindowAttributes = GetProcAddress(hUser32, "SetLayeredWindowAttributes")) != NULL &&

00070         g_fLayeredWindowInitDone = TRUE;

00073         g_pfnSetLayeredWindowAttributes = NULL;
00074         g_pfnUpdateLayeredWindow = NULL;
