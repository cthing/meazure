33 /// The window region is composed of single pixel rectangles arranged in a line from the start point to the end
34 /// point. The location of each rectangle is determined using the Bresenham algorithm adapted from
35 /// "Graphics Gems", Academic Press, 1990, p. 685. The line needs to be created in this brute force way because
36 /// relying on the polygon region method produces a horrible looking line.
37 ///
38 /// @param start [in] Start point for the line
39 /// @param end [in] End point for the line
40 /// @param addPoint [in] Function called to record the plotted point (x, y)
41 ///
42 inline void PlotLine(const POINT& start, const POINT& end, std::function<void(int, int)> addPoint) {
43 const int dx = end.x - start.x;
44 const int dy = end.y - start.y;
45
46 const int ax = 2 * std::abs(dx);
47 const int ay = 2 * std::abs(dy);
48
49 const int sx = (dx < 0) ? -1 : 1;
50 const int sy = (dy < 0) ? -1 : 1;
51
52 const auto [endx, endy] = end;
53
54 auto [x, y] = start;
55
56 if (ax > ay) { /* x dominant */
57 int d = ay - (ax / 2);
58 while (x != endx) {
59 if (d >= 0) {
60 y += sy;
61 d -= ax;
62 }
63 x += sx;
64 d += ay;
65 addPoint(x, y);
66 }
67 } else { /* y dominant */
68 int d = ax - (ay / 2);
69 while (y != endy) {
70 if (d >= 0) {
71 x += sx;
72 d -= ay;
73 }
74 y += sy;
75 d += ax;
76 addPoint(x, y);
77 }
78 }
79 }

81 /// The circular window region is composed of single pixel rectangles arranged in a circle. The location of each
82 /// rectangle is determined using an algorithm from the paper "A Fast Bresenham Type Algorithm for Drawing Circles"
83 /// by John Kennedy, Mathematics Dept., Santa Monica College (http://homepage.smc.edu/kennedy_john/BCIRCLE.PDF,
84 /// rkennedy@ix.netcom.com).
85 ///
86 /// @param center [in] Center of the circle
87 /// @param radius [in] Radius of the circle, in pixels
88 /// @param addPoint [in] Function called to record the plotted point (x, y)
89 ///
90 inline void PlotCircle(const POINT& center, int radius, std::function<void(int, int)> addPoint) {
91 const auto [xc, yc] = center;
92 int x = radius;
93 int y = 0;
94 int deltax = 1 - 2 * radius;
95 int deltay = 1;
96 int radiusError = 0;
97
98 while (x >= y) {
99 addPoint(xc + x, yc + y); // Octant 1
100 addPoint(xc - x, yc + y); // Octant 4
101 addPoint(xc - x, yc - y); // Octant 5
102 addPoint(xc + x, yc - y); // Octant 8
103 addPoint(xc + y, yc + x); // Octant 2
104 addPoint(xc - y, yc + x); // Octant 3
105 addPoint(xc - y, yc - x); // Octant 6
106 addPoint(xc + y, yc - x); // Octant 7
107
108 y++;
109 radiusError += deltay;
110 deltay += 2;
111
112 if ((2 * radiusError + deltax) > 0) {
113 x--;
114 radiusError += deltax;
115 deltax += 2;
116 }
117 }
118 }

120 /// Plots the points comprising the crosshair. Four petals comprise the top, bottom, left and right of the
121 /// crosshair. Each petal of the crosshair is made up of stacked rectangles. Each rectangle is thk high by
122 /// 2 * spread wide. Each rectangle is called a layer.
123 /// <pre>
124 /// |---| spread
125 /// ********* ---------
126 /// * * - thk |
127 /// * * |
128 /// * * |
129 /// * * |
130 /// * * 5 layers
131 /// * * |
132 /// * * |
133 /// * |
134 /// * -------------
135 /// </pre>
136 ///
137 /// @param size [in] total size of the crosshair
138 /// @param spread [in] half the width of the petal at its widest point
139 /// @param layers [in] number of rectangles comprising a crosshair petal
140 /// @param addPoint [in] Function called to record the plotted point (x, y)
141 ///
142 inline void PlotCrosshair(const SIZE& size, const SIZE& spread, int layers, std::function<void(int, int)> addPoint) {
143 const int xc = size.cx / 2;
144 const int yc = size.cy / 2;
145 const int thkx = xc / layers;
146 const int thky = yc / layers;
147
148 for (int layer = 0, spreadx = spread.cx, thk = 0; layer < layers && spreadx >= 0; layer++, spreadx--, thk += thky) {
149 addPoint(xc - spreadx, thk);
150 addPoint(xc + spreadx + 1, thk);
151 addPoint(xc + spreadx + 1, thk + thky);
152 addPoint(xc - spreadx, thk + thky);
153 }
154
155 //
156 // Left petal
157 //
158 for (int layer = 0, spready = spread.cy, thk = 0; layer < layers && spready >= 0; layer++, spready--, thk += thkx) {
159 addPoint(thk, yc - spready);
160 addPoint(thk, yc + spready + 1);
161 addPoint(thk + thkx, yc + spready + 1);
162 addPoint(thk + thkx, yc - spready);
163 }
164
165 //
166 // Bottom petal
167 //
168 for (int layer = 0, spreadx = spread.cx, thk = size.cy; layer < layers && spreadx >= 0; layer++, spreadx--, thk -= thky) {
169 addPoint(xc - spreadx, thk);
170 addPoint(xc + spreadx + 1, thk);
171 addPoint(xc + spreadx + 1, thk - thky);
172 addPoint(xc - spreadx, thk - thky);
173 }
174
175 //
176 // Right petal
177 //
178 for (int layer = 0, spready = spread.cy, thk = size.cx; layer < layers && spready >= 0; layer++, spready--, thk -= thkx) {
179 addPoint(thk, yc - spready);
180 addPoint(thk, yc + spready + 1);
181 addPoint(thk - thkx, yc + spready + 1);
182 addPoint(thk - thkx, yc - spready);
183 }
184 }
185};
